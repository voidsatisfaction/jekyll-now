(define (pow a b)
  (cond ((= b 0) 1)
        ((= (modulo b 2) 0)
         (let ((val (pow a (/ b 2))))
           (* val val)))
        (else
         (let ((val (pow a (/ (- b 1) 2))))
           (* val val a)))))

(define (n-digit-count a n cnt)
  (cond ((> a 9) cnt)
        ((> (pow 10 (- n 1)) (pow a n)) (n-digit-count (+ a 1) 1 cnt))
        (else (n-digit-count a (+ n 1) (+ cnt 1)))))
